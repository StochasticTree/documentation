17static inline void ExtractMultipleFeaturesFromMemory(std::vector<std::string>* text_data, const Parser* parser,
18 std::vector<int32_t>& column_indices, Eigen::MatrixXd& data,
19 data_size_t num_rows) {
20 std::vector<std::pair<int, double>> oneline_features;
21 auto& ref_text_data = *text_data;
22 int feature_counter;
23 bool column_matched;
24 for (data_size_t i = 0; i < num_rows; ++i) {
25 // unpack the vector of textlines read from file into a vector of (int, double) tuples
26 oneline_features.clear();
27 parser->ParseOneLine(ref_text_data[i].c_str(), &oneline_features);
28
29 // free processed line:
30 ref_text_data[i].clear();
31
32 // unload the data from oneline_features vector into the dataset variables containers
33 int feature_counter = 0;
34 for (auto& inner_data : oneline_features) {
35 int feature_idx = inner_data.first;
36 column_matched = (std::find(column_indices.begin(), column_indices.end(), feature_idx)
37 != column_indices.end());
38 if (column_matched){
39 data(i, feature_counter) = inner_data.second;
40 feature_counter += 1;
41 }
42 }
43 }
44 // free text data after use
45 text_data->clear();
46}

271 void UpdateBasis(double* data_ptr, data_size_t num_row, int num_col, bool is_row_major) {
272 CHECK(has_basis_);
273 CHECK_EQ(num_col, num_basis_);
274 // Copy data from R / Python process memory to Eigen matrix
275 double temp_value;
276 for (data_size_t i = 0; i < num_row; ++i) {
277 for (int j = 0; j < num_col; ++j) {
278 if (is_row_major){
279 // Numpy 2-d arrays are stored in "row major" order
280 temp_value = static_cast<double>(*(data_ptr + static_cast<data_size_t>(num_col) * i + j));
281 } else {
282 // R matrices are stored in "column major" order
283 temp_value = static_cast<double>(*(data_ptr + static_cast<data_size_t>(num_row) * j + i));
284 }
285 basis_.SetElement(i, j, temp_value);
286 }
287 }
288 }
296 void UpdateVarWeights(double* data_ptr, data_size_t num_row, bool exponentiate = true) {

299 double temp_value;

301 if (exponentiate) temp_value = std::exp(static_cast<double>(*(data_ptr + i)));
302 else temp_value = static_cast<double>(*(data_ptr + i));
303 var_weights_.SetElement(i, temp_value);
304 }
