186 inline void PredictInplace(Eigen::MatrixXd& covariates, Eigen::MatrixXd& basis, std::vector<double> &output,
187 int tree_begin, int tree_end, data_size_t offset = 0) {
188 double pred;
189 CHECK_EQ(covariates.rows(), basis.rows());
190 CHECK_EQ(output_dimension_, trees_[0]->OutputDimension());
191 CHECK_EQ(output_dimension_, basis.cols());
192 data_size_t n = covariates.rows();
193 data_size_t total_output_size = n;
194 if (output.size() < total_output_size + offset) {
195 Log::Fatal("Mismatched size of prediction vector and training data");
196 }
197 for (data_size_t i = 0; i < n; i++) {
198 pred = 0.0;
199 for (size_t j = tree_begin; j < tree_end; j++) {
200 auto &tree = *trees_[j];
201 std::int32_t nidx = EvaluateTree(tree, covariates, i);
202 for (int32_t k = 0; k < output_dimension_; k++) {
203 pred += tree.LeafValue(nidx, k) * basis(i, k);
204 }
205 }
206 if (is_exponentiated_) output[i + offset] = std::exp(pred);
207 else output[i + offset] = pred;
208 }
209 }

238 inline void PredictRawInplace(ForestDataset& dataset, std::vector<double> &output,
239 int tree_begin, int tree_end, data_size_t offset = 0) {
240 double pred;
241 Eigen::MatrixXd covariates = dataset.GetCovariates();
242 CHECK_EQ(output_dimension_, trees_[0]->OutputDimension());
243 data_size_t n = covariates.rows();
244 data_size_t total_output_size = n * output_dimension_;
245 if (output.size() < total_output_size + offset) {
246 Log::Fatal("Mismatched size of raw prediction vector and training data");
247 }
248 for (data_size_t i = 0; i < n; i++) {
249 for (int32_t k = 0; k < output_dimension_; k++) {
250 pred = 0.0;
251 for (size_t j = tree_begin; j < tree_end; j++) {
252 auto &tree = *trees_[j];
253 int32_t nidx = EvaluateTree(tree, covariates, i);
254 pred += tree.LeafValue(nidx, k);
255 }
256 output[i*output_dimension_ + k + offset] = pred;
257 }
