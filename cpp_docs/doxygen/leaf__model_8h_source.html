208class GaussianConstantSuffStat {
209 public:
210 data_size_t n;
211 double sum_w;
212 double sum_yw;
213 GaussianConstantSuffStat() {
214 n = 0;
215 sum_w = 0.0;
216 sum_yw = 0.0;
217 }
218 void IncrementSuffStat(ForestDataset& dataset, Eigen::VectorXd& outcome, ForestTracker& tracker, data_size_t row_idx, int tree_idx) {
219 n += 1;
220 if (dataset.HasVarWeights()) {
221 sum_w += 1/dataset.VarWeightValue(row_idx);
222 sum_yw += outcome(row_idx, 0)/dataset.VarWeightValue(row_idx);
223 } else {
224 sum_w += 1.0;
225 sum_yw += outcome(row_idx, 0);
226 }
227 }
228 void ResetSuffStat() {
229 n = 0;
230 sum_w = 0.0;
231 sum_yw = 0.0;
232 }
233 void AddSuffStat(GaussianConstantSuffStat& lhs, GaussianConstantSuffStat& rhs) {
234 n = lhs.n + rhs.n;
235 sum_w = lhs.sum_w + rhs.sum_w;
236 sum_yw = lhs.sum_yw + rhs.sum_yw;
237 }
238 void SubtractSuffStat(GaussianConstantSuffStat& lhs, GaussianConstantSuffStat& rhs) {
239 n = lhs.n - rhs.n;
240 sum_w = lhs.sum_w - rhs.sum_w;
241 sum_yw = lhs.sum_yw - rhs.sum_yw;
242 }
243 bool SampleGreaterThan(data_size_t threshold) {
244 return n > threshold;
245 }
246 bool SampleGreaterThanEqual(data_size_t threshold) {
247 return n >= threshold;
248 }
249 data_size_t SampleSize() {
250 return n;
251 }
252};

363class GaussianMultivariateRegressionSuffStat {
364 public:
365 data_size_t n;
366 int p;
367 Eigen::MatrixXd XtWX;
368 Eigen::MatrixXd ytWX;
369 GaussianMultivariateRegressionSuffStat(int basis_dim) {
370 n = 0;
371 XtWX = Eigen::MatrixXd::Zero(basis_dim, basis_dim);
372 ytWX = Eigen::MatrixXd::Zero(1, basis_dim);
373 p = basis_dim;
374 }
375 void IncrementSuffStat(ForestDataset& dataset, Eigen::VectorXd& outcome, ForestTracker& tracker, data_size_t row_idx, int tree_idx) {
376 n += 1;
377 if (dataset.HasVarWeights()) {
378 XtWX += dataset.GetBasis()(row_idx, Eigen::all).transpose()*dataset.GetBasis()(row_idx, Eigen::all)/dataset.VarWeightValue(row_idx);
379 ytWX += (outcome(row_idx, 0)*(dataset.GetBasis()(row_idx, Eigen::all)))/dataset.VarWeightValue(row_idx);
380 } else {
381 XtWX += dataset.GetBasis()(row_idx, Eigen::all).transpose()*dataset.GetBasis()(row_idx, Eigen::all);
382 ytWX += (outcome(row_idx, 0)*(dataset.GetBasis()(row_idx, Eigen::all)));
383 }
384 }
385 void ResetSuffStat() {
386 n = 0;
387 XtWX = Eigen::MatrixXd::Zero(p, p);
388 ytWX = Eigen::MatrixXd::Zero(1, p);

391 n = lhs.n + rhs.n;

394 }

396 n = lhs.n - rhs.n;

506static inline SuffStatVariant suffStatFactory(ModelType model_type, int basis_dim = 0) {
507 if (model_type == kConstantLeafGaussian) {
508 return createSuffStat<GaussianConstantSuffStat>();
509 } else if (model_type == kUnivariateRegressionLeafGaussian) {

518static inline LeafModelVariant leafModelFactory(ModelType model_type, double tau, Eigen::MatrixXd& Sigma0, double a, double b) {
519 if (model_type == kConstantLeafGaussian) {
520 return createLeafModel<GaussianConstantLeafModel, double>(tau);
521 } else if (model_type == kUnivariateRegressionLeafGaussian) {

523 } else if (model_type == kMultivariateRegressionLeafGaussian) {

526 return createLeafModel<LogLinearVarianceLeafModel, double, double>(a, b);
527 } else {
528 Log::Fatal("Incompatible model type provided to leaf model factory");
529 }

532template<typename SuffStatType>
533static inline void AccumulateSuffStatProposed(SuffStatType& node_suff_stat, SuffStatType& left_suff_stat, SuffStatType& right_suff_stat, ForestDataset& dataset, ForestTracker& tracker,
534 ColumnVector& residual, double global_variance, TreeSplit& split, int tree_num, int leaf_num, int split_feature) {
535 // Acquire iterators
536 auto node_begin_iter = tracker.UnsortedNodeBeginIterator(tree_num, leaf_num);
537 auto node_end_iter = tracker.UnsortedNodeEndIterator(tree_num, leaf_num);
538
539 // Accumulate sufficient statistics
540 for (auto i = node_begin_iter; i != node_end_iter; i++) {
541 auto idx = *i;
542 double feature_value = dataset.CovariateValue(idx, split_feature);
543 node_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
544 if (split.SplitTrue(feature_value)) {
545 left_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
546 } else {
547 right_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
548 }
549 }
550}

552template<typename SuffStatType>
553static inline void AccumulateSuffStatExisting(SuffStatType& node_suff_stat, SuffStatType& left_suff_stat, SuffStatType& right_suff_stat, ForestDataset& dataset, ForestTracker& tracker,
554 ColumnVector& residual, double global_variance, int tree_num, int split_node_id, int left_node_id, int right_node_id) {
555 // Acquire iterators
556 auto left_node_begin_iter = tracker.UnsortedNodeBeginIterator(tree_num, left_node_id);
557 auto left_node_end_iter = tracker.UnsortedNodeEndIterator(tree_num, left_node_id);
558 auto right_node_begin_iter = tracker.UnsortedNodeBeginIterator(tree_num, right_node_id);
559 auto right_node_end_iter = tracker.UnsortedNodeEndIterator(tree_num, right_node_id);
560
561 // Accumulate sufficient statistics for the left and split nodes
562 for (auto i = left_node_begin_iter; i != left_node_end_iter; i++) {
563 auto idx = *i;
564 left_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
565 node_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
566 }
567
568 // Accumulate sufficient statistics for the right and split nodes
569 for (auto i = right_node_begin_iter; i != right_node_end_iter; i++) {
570 auto idx = *i;
571 right_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
572 node_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
573 }

576template<typename SuffStatType, bool sorted>
577static inline void AccumulateSingleNodeSuffStat(SuffStatType& node_suff_stat, ForestDataset& dataset, ForestTracker& tracker, ColumnVector& residual, int tree_num, int node_id) {
578 // Acquire iterators
579 std::vector<data_size_t>::iterator node_begin_iter;
580 std::vector<data_size_t>::iterator node_end_iter;
581 if (sorted) {
582 // Default to the first feature if we're using the presort tracker
583 node_begin_iter = tracker.SortedNodeBeginIterator(node_id, 0);
584 node_end_iter = tracker.SortedNodeEndIterator(node_id, 0);
585 } else {
586 node_begin_iter = tracker.UnsortedNodeBeginIterator(tree_num, node_id);
587 node_end_iter = tracker.UnsortedNodeEndIterator(tree_num, node_id);
588 }
589
590 // Accumulate sufficient statistics
591 for (auto i = node_begin_iter; i != node_end_iter; i++) {
592 auto idx = *i;
593 node_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);

597template<typename SuffStatType>
598static inline void AccumulateCutpointBinSuffStat(SuffStatType& left_suff_stat, ForestTracker& tracker, CutpointGridContainer& cutpoint_grid_container,
599 ForestDataset& dataset, ColumnVector& residual, double global_variance, int tree_num, int node_id,
600 int feature_num, int cutpoint_num) {
601 // Acquire iterators
602 auto node_begin_iter = tracker.SortedNodeBeginIterator(node_id, feature_num);
603 auto node_end_iter = tracker.SortedNodeEndIterator(node_id, feature_num);
604
605 // Determine node start point
606 data_size_t node_begin = tracker.SortedNodeBegin(node_id, feature_num);
607
608 // Determine cutpoint bin start and end points
609 data_size_t current_bin_begin = cutpoint_grid_container.BinStartIndex(cutpoint_num, feature_num);
610 data_size_t current_bin_size = cutpoint_grid_container.BinLength(cutpoint_num, feature_num);
611 data_size_t next_bin_begin = cutpoint_grid_container.BinStartIndex(cutpoint_num + 1, feature_num);
612
613 // Cutpoint specific iterators
614 // TODO: fix the hack of having to subtract off node_begin, probably by cleaning up the CutpointGridContainer interface
615 auto cutpoint_begin_iter = node_begin_iter + (current_bin_begin - node_begin);
616 auto cutpoint_end_iter = node_begin_iter + (next_bin_begin - node_begin);
617
618 // Accumulate sufficient statistics
619 for (auto i = cutpoint_begin_iter; i != cutpoint_end_iter; i++) {
620 auto idx = *i;
621 left_suff_stat.IncrementSuffStat(dataset, residual.GetData(), tracker, idx, tree_num);
622 }
