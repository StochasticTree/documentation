119 void ChangeToLeaf(std::int32_t nid, double value) {
120 CHECK(this->IsLeaf(this->LeftChild(nid)));
121 CHECK(this->IsLeaf(this->RightChild(nid)));
122 this->DeleteNode(this->LeftChild(nid));
123 this->DeleteNode(this->RightChild(nid));
124 this->SetLeaf(nid, value);

126 // Add nid to leaves and remove from internal nodes and leaf parents (if it was there)
127 leaves_.push_back(nid);

129 internal_nodes_.erase(std::remove(internal_nodes_.begin(), internal_nodes_.end(), nid), internal_nodes_.end());

131 // Check if the other child of nid's parent node is also a leaf, if so, add parent back to leaf parents

204 template <typename Func> void WalkTree(Func func) const {
205 std::stack<std::int32_t> nodes;
206 nodes.push(kRoot);
207 auto &self = *this;
208 while (!nodes.empty()) {
209 auto nidx = nodes.top();
210 nodes.pop();
211 if (!func(nidx)) {
212 return;
213 }
214 auto left = self.LeftChild(nidx);
215 auto right = self.RightChild(nidx);
216 if (left != Tree::kInvalidNodeId) {
217 nodes.push(left);
218 }
219 if (right != Tree::kInvalidNodeId) {
220 nodes.push(right);
221 }
222 }
223 }

346 std::int32_t MaxLeafDepth() const {
347 std::int32_t max_depth = 0;
348 std::stack<std::int32_t> nodes;
349 std::stack<std::int32_t> node_depths;
350 nodes.push(kRoot);
351 node_depths.push(0);
352 auto &self = *this;
353 while (!nodes.empty()) {
354 auto nidx = nodes.top();
355 nodes.pop();
356 auto node_depth = node_depths.top();
357 node_depths.pop();
358 bool valid_node = !self.IsDeleted(nidx);
359 if (valid_node) {
360 if (node_depth > max_depth) max_depth = node_depth;
361 auto left = self.LeftChild(nidx);
362 auto right = self.RightChild(nidx);
363 if (left != Tree::kInvalidNodeId) {
364 nodes.push(left);
365 node_depths.push(node_depth+1);
366 }
367 if (right != Tree::kInvalidNodeId) {
368 nodes.push(right);
369 node_depths.push(node_depth+1);
370 }
371 }
372 }
373 return max_depth;
374 }

397 if (output_dimension_ == 1) {

497 [[nodiscard]] bool IsLeafParent(std::int32_t nid) const {
498 // False until we deduce left and right node are
499 // available and both are leaves
500 bool is_left_leaf = false;
501 bool is_right_leaf = false;
502 // Check if node nidx is a leaf, if so, return false
503 bool is_leaf = this->IsLeaf(nid);
504 if (is_leaf){
505 return false;
506 } else {
507 // If nidx is not a leaf, it must have left and right nodes
508 // so we check if those are leaves
509 std::int32_t left_node = LeftChild(nid);
510 std::int32_t right_node = RightChild(nid);
511 is_left_leaf = IsLeaf(left_node);
512 is_right_leaf = IsLeaf(right_node);
513 }
514 return is_left_leaf && is_right_leaf;
515 }

808inline bool SplitTrueCategorical(double fvalue, std::vector<std::uint32_t> const& category_list) {
809 bool category_matched;
810 // A valid (integer) category must satisfy two criteria:
811 // 1) it must be exactly representable as double
812 // 2) it must fit into uint32_t
813 auto max_representable_int
814 = std::min(static_cast<double>(std::numeric_limits<std::uint32_t>::max()),
815 static_cast<double>(std::uint64_t(1) << std::numeric_limits<double>::digits));
816 if (fvalue < 0 || std::fabs(fvalue) > max_representable_int) {
817 category_matched = false;
818 } else {
819 auto const category_value = static_cast<std::uint32_t>(fvalue);
820 category_matched = (std::find(category_list.begin(), category_list.end(), category_value)
821 != category_list.end());
822 }
823 return category_matched;
824}

853inline int EvaluateTree(Tree const& tree, Eigen::MatrixXd& data, int row) {
854 int node_id = 0;
855 while (!tree.IsLeaf(node_id)) {
856 auto const split_index = tree.SplitIndex(node_id);
857 double const fvalue = data(row, split_index);
858 if (std::isnan(fvalue)) {
859 node_id = tree.DefaultChild(node_id);
860 } else {
861 if (tree.NodeType(node_id) == StochTree::TreeNodeType::kCategoricalSplitNode) {
862 node_id = NextNodeCategorical(fvalue, tree.CategoryList(node_id),
863 tree.LeftChild(node_id), tree.RightChild(node_id));
864 } else {
865 node_id = NextNodeNumeric(fvalue, tree.Threshold(node_id), tree.LeftChild(node_id), tree.RightChild(node_id));
866 }
867 }
868 }
869 return node_id;
870}
